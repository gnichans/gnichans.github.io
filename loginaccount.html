document.getElementById('loginForm').addEventListener('submit', function(event) { event.preventDefault(); // Ngăn việc reload trang const username = document.getElementById('username').value.trim(); const password = document.getElementById('password').value.trim(); const errorMsg = document.getElementById('errorMsg'); // Phần hiển thị lỗi // Xóa thông báo lỗi cũ errorMsg.textContent = ''; errorMsg.style.display = 'none'; // Kiểm tra các trường dữ liệu if (!username || !password) { errorMsg.textContent = 'Hãy nhập thông tin đầy đủ!'; errorMsg.style.display = 'block'; return; } // Kiểm tra thông tin đăng nhập const existingUsers = JSON.parse(localStorage.getItem('users')) || []; const user = existingUsers.find(user => user.username === username && user.password === password); if (user) { alert('Đăng nhập thành công!'); window.location.href = `/home?logined&user=${encodeURIComponent(username)}`; } else { errorMsg.textContent = 'Tên đăng nhập hoặc mật khẩu không chính xác.'; errorMsg.style.display = 'block'; } });
